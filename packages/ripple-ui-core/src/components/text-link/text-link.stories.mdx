import {
  Canvas,
  Meta,
  Story
} from '@storybook/addon-docs'
import { ArgsTable } from '@storybook/addon-docs'
import RplTextLink from './text-link.vue'
import RplIcon from '../icon/icon.vue'

import { expect } from '@storybook/jest'
import { within, userEvent } from '@storybook/testing-library'
import { rplEventBus } from './../../index.ts'
const playFunction = async ({ canvasElement }) => {
  let fired = false
  rplEventBus.all.clear()
  const handler = () => { fired = !fired }
  rplEventBus.on('rpl-text-link/click', handler)
  const canvas = await within(canvasElement)
  await userEvent.click(canvas.getByRole('link'))
  await expect(fired).toBeTruthy()
  rplEventBus.off('rpl-text-link/click', handler)
}

export const Template = (args) => ({
  components: { RplTextLink },
  setup() {
    return { args }
  },
  template: '<p class="rpl-type-body">This para contains a <RplTextLink @click.prevent="args.onClick()" v-bind="args">link</RplTextLink> in the middle.</p>'
})
