import {
  Canvas,
  Meta,
  Story
} from '@storybook/addon-docs'
import { ArgsTable } from '@storybook/addon-docs'
import RplTextLink from './text-link.vue'
import RplIcon from '../icon/icon.vue'
import { within, userEvent } from '@storybook/testing-library'
import { eventCheck, a11yCheck, a11yStoryCheck } from './../../../.storybook/interactions.js'
const eventTestFunction = async ({ canvasElement }) => {
  const canvas = await within(canvasElement)
  const link = await canvas.getByRole('link')
  await a11yCheck(canvasElement)
  await eventCheck(link, 'rpl-text-link/click')
}

export const Template = (args) => ({
  components: { RplTextLink },
  setup() {
    return { args }
  },
  template: '<p class="sb-container">This para contains a <RplTextLink @click.prevent="args.onClick()" v-bind="args">link</RplTextLink> in the middle.</p>'
})
