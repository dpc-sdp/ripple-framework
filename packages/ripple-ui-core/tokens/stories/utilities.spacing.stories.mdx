import { Meta, Canvas, Story } from '@storybook/addon-docs'
import { RplPaddingUtilities, RplMarginUtilities, RplSizeUtilities } from './constants'
import sbTokenTable from './../../.storybook/components/token-table.vue'
import sbTokenTableRow from './../../.storybook/components/token-row.vue'
import sbTokenTableCol from './../../.storybook/components/token-col.vue'
import sbTokenTableSwatch from './../../.storybook/components/token-swatch.vue'


export const PaddingTemplate = (args) => ({
  components: { sbTokenTable, sbTokenTableRow, sbTokenTableCol, sbTokenTableSwatch },
  setup() {
    return {
      columns: ['Style', 'Class'],
      RplPaddingUtilities,
      getItemStyles: (item) => {
        return `background-color: ${item.color}; display: inline-block;`
      }
    }
  },
  template: `
    <sbTokenTable :columns="columns" full>
      <sbTokenTableRow lined v-for="item in RplPaddingUtilities" :key="item">
        <sbTokenTableCol>
          <div :style="getItemStyles(item)" :className="item.class">
            {{item.title}}
          </div>
        </sbTokenTableCol>
        <sbTokenTableCol>.{{item.class}}</sbTokenTableCol>
      </sbTokenTableRow>
    </sbTokenTable>
  `
});
